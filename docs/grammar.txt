EBNF
----
Module
    = StmtList
    .

StmtList
    = { Statement }
    .

Statement
    = Declaration
    | Assignment
    | CallStmt
    | RETURN
    .

Declaration
    = [ AttributeList ]
    ( VAR
    | DECLARE
    | FUNCTION
    | SUB
    )
    .

AttributeList
    = "[" Attribute { ","  Attribute } "]"
    .

Attribute
    = IdentExpr [ AttributeArgList ]
    .

AttributeArgList
    = "=" Literal
    | "(" [ Literal { "," Literal } ] ")"
    .

VAR
    = "VAR" identifier
    ( "=" Expression
    | "AS" TypeExpr [ "=" Expression ]
    )
    .

DECLARE
    = "DECLARE" FuncSignature
    .

FuncSignature
    = "FUNCTION" id [ "(" [ FuncParamList ] ")" ] "AS" TypeExpr
    | "SUB" id [ "(" FuncParamList ")" ]
    .

FuncParamList
    = FuncParam { "," FuncParam } [ "," "..." ]
    | "..."
    .

FuncParam
    = id "AS" TypeExpr
    .

TypeExpr
    = id
    .
